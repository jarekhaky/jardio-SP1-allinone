// ========== UTILITY FUNCTIONS ==========
String shortenName(String name) {
  int dot = name.lastIndexOf('.');
  if (dot > 0) name = name.substring(0, dot);
  if (name.length() > 20) name = name.substring(0, 20) + "...";
  return name;
}

String urlEncode(String str) {
  String encoded = "";
  const char* hex = "0123456789ABCDEF";
  for (unsigned int i = 0; i < str.length(); i++) {
    unsigned char c = (unsigned char)str.charAt(i);
    if (isalnum(c) || c == '-' || c == '_' || c == '.' || c == '~') {
      encoded += (char)c;
    } else {
      encoded += '%';
      encoded += hex[(c >> 4) & 0x0F];
      encoded += hex[c & 0x0F];
    }
  }
  return encoded;
}

String urlDecode(String str) {
  String decoded = "";
  for (unsigned int i = 0; i < str.length(); i++) {
    if (str.charAt(i) == '%') {
      if (i + 2 < str.length()) {
        char c = strtol(str.substring(i + 1, i + 3).c_str(), NULL, 16);
        decoded += c;
        i += 2;
      }
    } else if (str.charAt(i) == '+') decoded += ' ';
    else decoded += str.charAt(i);
  }
  return decoded;
}
String generateEventId() {
  static unsigned long counter = 0;
  return "evt_" + String(millis()) + "_" + String(counter++);
}

String jsQuote(const String& s) {
  String out = "";
  for (unsigned int i = 0; i < s.length(); i++) {
    char c = s.charAt(i);
    if (c == '\\') out += "\\\\";
    else if (c == '\"') out += "\\\"";
    else if (c == '\'') out += "\\'";
    else if (c == '\n') out += "\\n";
    else if (c == '\r') out += "\\r";
    else out += c;
  }
  return out;
}

String htmlEscape(const String& s) {
  String out = "";
  for (unsigned int i = 0; i < s.length(); i++) {
    char c = s.charAt(i);
    if (c == '&') out += "&amp;";
    else if (c == '<') out += "&lt;";
    else if (c == '>') out += "&gt;";
    else if (c == '\"') out += "&quot;";
    else if (c == '\'') out += "&#39;";
    else out += c;
  }
  return out;
}

int getSpotPrefix(String filename) {
  int underscorePos = filename.indexOf('_');
  if (underscorePos <= 0) return -1;
  String prefix = filename.substring(0, underscorePos);
  int num = prefix.toInt();
  if (num >= 1 && num <= 10) return num;
  return -1;
}

String findSpotFile(int spotNum) {
  File root = SD.open("/spot");
  if (!root) return "";
  File f = root.openNextFile();
  while (f) {
    if (!f.isDirectory()) {
      String n = f.name();
      if (n.startsWith("/")) n = n.substring(1);
      int slash = n.lastIndexOf('/');
      if (slash >= 0) n = n.substring(slash + 1);
      if (getSpotPrefix(n) == spotNum) {
        String result = n;
        f.close();
        root.close();
        return result;
      }
    }
    f = root.openNextFile();
  }
  root.close();
  return "";
}

void initPlaylists() {
  if (!SD.exists("/playlist")) SD.mkdir("/playlist");
  for (int i = 1; i <= 10; i++) {
    String path = "/playlist/pl" + String(i) + ".csv";
    if (!SD.exists(path)) {
      File f = SD.open(path, FILE_WRITE);
      if (f) f.close();
    }
    lastTrackIndexPerPlaylist[i] = -1;
  }
}

bool isInPlaylist(const String& filename, int playlistNum) {
  String path = "/playlist/pl" + String(playlistNum) + ".csv";
  File f = SD.open(path);
  if (!f) return false;
  while (f.available()) {
    String line = f.readStringUntil('\n');
    line.trim();
    if (line == filename) {
      f.close();
      return true;
    }
  }
  f.close();
  return false;
}

int getPlaylistTrackCount(int playlistNum) {
  if (playlistNum < 1 || playlistNum > 10) return 0;
  String path = "/playlist/pl" + String(playlistNum) + ".csv";
  File f = SD.open(path);
  if (!f) return 0;
  int count = 0;
  while (f.available()) {
    String line = f.readStringUntil('\n');
    line.trim();
    if (line.length() > 0) count++;
  }
  f.close();
  return count;
}

String getPlaylistTrackAt(int playlistNum, int index) {
  if (playlistNum < 1 || playlistNum > 10 || index < 0) return "";
  String path = "/playlist/pl" + String(playlistNum) + ".csv";
  File f = SD.open(path);
  if (!f) return "";
  int i = 0;
  while (f.available()) {
    String line = f.readStringUntil('\n');
    line.trim();
    if (line.length() == 0) continue;
    if (i == index) {
      f.close();
      return line;
    }
    i++;
  }
  f.close();
  return "";
}

bool playTrackFromCurrentPlaylistIndex(int index) {
  if (currentPlaylistNum < 1 || currentPlaylistNum > 10) return false;
  String t = getPlaylistTrackAt(currentPlaylistNum, index);
  if (t.length() == 0) return false;

  currentPlaylistTrackIndex = index;
  lastTrackIndexPerPlaylist[currentPlaylistNum] = index;
  nowPlayingTrack = t;
  lastPlayedTrackName = t;
  lastPlayedFile = "/music/" + t;
  currentPlaylistName = "PL" + String(currentPlaylistNum);
  isMusicPlaying = true;
  manualDirectPlayback = false;
  userPaused = false;
  userStopped = false;
  audio.setVolume(currentMusicVolume);
  audio.connecttoFS(SD, lastPlayedFile.c_str());

  Serial.printf("[PLAYLIST] PL%d track #%d: %s (vol=%d)\n",
                currentPlaylistNum, index + 1, t.c_str(), currentMusicVolume);
  return true;
}

bool playNextTrackInCurrentPlaylist(bool wrapIfNeeded) {
  if (currentPlaylistNum < 1 || currentPlaylistNum > 10) return false;
  int total = getPlaylistTrackCount(currentPlaylistNum);
  if (total <= 0) return false;

  int nextIndex = -1;
  bool sh = playlistShuffle[currentPlaylistNum];

  if (sh) {
    if (!rngSeedReady) {
      randomSeed((uint32_t)esp_random());
      rngSeedReady = 1;
    }
    if (total == 1) {
      nextIndex = 0;
    } else {
      int guard = 0;
      do {
        nextIndex = random(0, total);
        guard++;
      } while (nextIndex == currentPlaylistTrackIndex && guard < 20);
      if (nextIndex == currentPlaylistTrackIndex) {
        nextIndex = (currentPlaylistTrackIndex + 1) % total;
      }
    }
  } else {
    if (currentPlaylistTrackIndex < 0) {
      int start = playlistStartIndex[currentPlaylistNum];
      if (start < 0) start = 0;
      if (start >= total) start = 0;
      nextIndex = start;
    } else {
      nextIndex = currentPlaylistTrackIndex + 1;
      if (nextIndex >= total) {
        if (!wrapIfNeeded) return false;
        nextIndex = 0;
      }
    }
  }

  // anti-repeat by name (defensive, mainly for duplicates in csv)
  String candidate = getPlaylistTrackAt(currentPlaylistNum, nextIndex);
  if (total > 1 && candidate == lastPlayedTrackName) {
    int alt = (nextIndex + 1) % total;
    String altName = getPlaylistTrackAt(currentPlaylistNum, alt);
    if (altName.length() > 0 && altName != lastPlayedTrackName) nextIndex = alt;
  }

  return playTrackFromCurrentPlaylistIndex(nextIndex);
}

void resetEventRuntimeState(ScheduleEvent& evt) {
  evt.active = false;
  evt.spotRepeatCount = 0;

  evt.lastTriggerMinute = -1;
}

bool parseTimeString(const String& value, int& hour, int& minute) {
  if (value.length() != 5 || value.charAt(2) != ':') return false;
  if (!isDigit(value.charAt(0)) || !isDigit(value.charAt(1)) || !isDigit(value.charAt(3)) || !isDigit(value.charAt(4))) return false;
  hour = value.substring(0, 2).toInt();
  minute = value.substring(3, 5).toInt();
  return hour >= 0 && hour <= 23 && minute >= 0 && minute <= 59;
}

int timeToMinutes(const String& value) {
  int hour = 0, minute = 0;
  if (!parseTimeString(value, hour, minute)) return -1;
  return hour * 60 + minute;
}

bool isValidTimeRangeForEvent(const ScheduleEvent& evt) {
  int start = timeToMinutes(evt.startTime);
  if (start < 0) return false;
  if (evt.type == "spot") return true;
  int end = timeToMinutes(evt.endTime);
  return end >= start;
}

bool isEventActiveNow(const ScheduleEvent& evt, int currentDay, int currentMinute) {
  if (evt.day != currentDay) return false;
  int start = timeToMinutes(evt.startTime);
  if (start < 0) return false;
  if (evt.type == "spot") return currentMinute == start;
  int end = timeToMinutes(evt.endTime);
  if (end < start) return false;
  return currentMinute >= start && currentMinute <= end;
}

int findActivePlaylistForNow(int currentDay, int currentMinute) {
  int playlistNum = -1;
  for (int i = 0; i < scheduleEventCount; i++) {
    const ScheduleEvent& evt = scheduleEvents[i];
    if (evt.type != "playlist") continue;
    if (isEventActiveNow(evt, currentDay, currentMinute)) {
      playlistNum = evt.target;
    }
  }
  return playlistNum;
}

bool eventsEquivalentForDay(const ScheduleEvent& a, const ScheduleEvent& b, int targetDay) {
  if (a.type != b.type) return false;
  if (b.day != targetDay) return false;
  if (a.startTime != b.startTime) return false;
  if (a.type == "playlist") return a.endTime == b.endTime && a.target == b.target;
  if (a.type == "volume") return a.endTime == b.endTime && a.volume == b.volume;
  if (a.type == "spot") return a.target == b.target && a.repeat == b.repeat && a.volume == b.volume;
  return false;
}

String getEventTimeLabel(const ScheduleEvent& evt) {
  if (evt.type == "spot") return evt.startTime;
  return evt.startTime + "-" + evt.endTime;
}

String getEventSummary(const ScheduleEvent& evt) {
  if (evt.type == "playlist") return getEventTimeLabel(evt) + " PL" + String(evt.target);
  if (evt.type == "volume") return getEventTimeLabel(evt) + " VOL " + String(evt.volume);
  return evt.startTime + " Spot " + String(evt.target) + " " + String(evt.repeat) + "x V" + String(evt.volume);
}
void loadScheduleEvents() {
  scheduleEventCount = 0;
  File f = SD.open("/scheduler/events.json");
  if (!f) {
    Serial.println("[SCHEDULER] No events file found");
    return;
  }

  size_t fileSize = f.size();
  Serial.printf("[SCHEDULER] Loading /scheduler/events.json (%u bytes)\n", (unsigned int)fileSize);

  DynamicJsonDocument doc(32768);
  DeserializationError err = deserializeJson(doc, f);
  f.close();

  if (err) {
    Serial.printf("[SCHEDULER] Failed to parse events.json: %s\n", err.c_str());
    return;
  }

  JsonArray arr;
  if (doc.is<JsonArray>()) arr = doc.as<JsonArray>();
  else if (doc.is<JsonObject>() && doc["events"].is<JsonArray>()) arr = doc["events"].as<JsonArray>();
  else {
    Serial.println("[SCHEDULER] Unsupported events root");
    return;
  }

  int migratedCount = 0, skippedCount = 0, index = 0;
  for (JsonVariant item : arr) {
    if (scheduleEventCount >= 100) break;
    if (!item.is<JsonObject>()) {
      Serial.printf("[SCHEDULER] Skipping non-object event at index %d\n", index++);
      skippedCount++;
      continue;
    }

    JsonObject obj = item.as<JsonObject>();
    ScheduleEvent& evt = scheduleEvents[scheduleEventCount];
    evt.id = obj["id"] | generateEventId();
    evt.day = obj["day"] | 0;
    evt.type = obj["type"].as<String>();
    String legacyTime = obj["time"] | "";
    evt.startTime = obj["startTime"] | legacyTime;
    evt.endTime = obj["endTime"] | ((evt.type == "spot") ? "" : evt.startTime);
    evt.target = obj["target"] | 1;
    evt.repeat = obj["repeat"] | 1;
    evt.volume = obj["volume"] | ((evt.type == "spot") ? defaultSpotVolume : defaultMusicVolume);
    evt.repeat = constrain(evt.repeat, 1, 5);
    evt.volume = constrain(evt.volume, 0, 21);
    resetEventRuntimeState(evt);

    if (evt.day < 0 || evt.day > 6) {
      skippedCount++;
      index++;
      continue;
    }
    if (evt.type != "playlist" && evt.type != "volume" && evt.type != "spot") {
      skippedCount++;
      index++;
      continue;
    }

    if (legacyTime.length() > 0 && !obj.containsKey("startTime")) migratedCount++;
    if (evt.type == "spot") {
      evt.endTime = "";
      if (evt.target < 1 || evt.target > 10) evt.target = 1;
    } else if (evt.type == "playlist") {
      if (evt.target < 1 || evt.target > 10) evt.target = 1;
    } else {
      evt.target = 0;
    }

    if (!isValidTimeRangeForEvent(evt)) {
      skippedCount++;
      index++;
      continue;
    }

    scheduleEventCount++;
    index++;
  }
  Serial.printf("[SCHEDULER] Loaded %d events (migrated=%d, skipped=%d)\n", scheduleEventCount, migratedCount, skippedCount);
}

void saveScheduleEvents() {
  const char* path = "/scheduler/events.json";
  const char* tmpPath = "/scheduler/events.tmp";
  SD.remove(tmpPath);

  File f = SD.open(tmpPath, FILE_WRITE);
  if (!f) {
    Serial.println("[SCHEDULER] Failed to open temp events file");
    return;
  }

  DynamicJsonDocument doc(32768);
  doc["version"] = FIRMWARE_VERSION;
  JsonArray arr = doc.createNestedArray("events");

  for (int i = 0; i < scheduleEventCount; i++) {
    JsonObject obj = arr.createNestedObject();
    obj["id"] = scheduleEvents[i].id;
    obj["day"] = scheduleEvents[i].day;
    obj["startTime"] = scheduleEvents[i].startTime;
    if (scheduleEvents[i].type != "spot") obj["endTime"] = scheduleEvents[i].endTime;
    obj["time"] = scheduleEvents[i].startTime;  // legacy
    obj["type"] = scheduleEvents[i].type;
    obj["target"] = scheduleEvents[i].target;
    obj["repeat"] = scheduleEvents[i].repeat;
    obj["volume"] = scheduleEvents[i].volume;
  }

  size_t written = serializeJson(doc, f);
  f.flush();
  size_t tempSize = f.size();
  f.close();

  if (written == 0) {
    Serial.println("[SCHEDULER] Failed to serialize events");
    SD.remove(tmpPath);
    return;
  }

  if (SD.exists(path)) SD.remove(path);
  bool renamed = SD.rename(tmpPath, path);
  if (!renamed) {
    File src = SD.open(tmpPath, FILE_READ);
    File dst = SD.open(path, FILE_WRITE);
    if (!src || !dst) {
      if (src) src.close();
      if (dst) dst.close();
      SD.remove(tmpPath);
      return;
    }
    uint8_t buffer[256];
    while (src.available()) {
      int len = src.read(buffer, sizeof(buffer));
      if (len > 0) dst.write(buffer, len);
    }
    dst.flush();
    src.close();
    dst.close();
    SD.remove(tmpPath);
  }

  File verify = SD.open(path, FILE_READ);
  size_t finalSize = verify ? verify.size() : 0;
  if (verify) verify.close();
  Serial.printf("[SCHEDULER] Saved %d events (%u bytes temp, %u bytes final)\n",
                scheduleEventCount, (unsigned int)tempSize, (unsigned int)finalSize);
}

void playPlaylist(int playlistNum, bool resetIndex) {
  if (playlistNum < 1 || playlistNum > 10) return;

  if (currentPlaylistNum != playlistNum) resetIndex = true;
  currentPlaylistNum = playlistNum;
  currentPlaylistName = "PL" + String(playlistNum);
  isMusicPlaying = true;
  manualDirectPlayback = false;
  userPaused = false;
  userStopped = false;

  int total = getPlaylistTrackCount(playlistNum);
  if (total <= 0) {
    Serial.printf("[PLAYLIST] PL%d empty\n", playlistNum);
    isMusicPlaying = false;
    nowPlayingTrack = "";
    currentPlaylistTrackIndex = -1;
    return;
  }

  if (resetIndex) {
    if (playlistShuffle[playlistNum]) currentPlaylistTrackIndex = -1;
    else {
      int start = playlistStartIndex[playlistNum];
      if (start < 0 || start >= total) start = 0;
      currentPlaylistTrackIndex = start - 1;  // so next call starts at start
    }
  }

  if (!playNextTrackInCurrentPlaylist(true)) {
    Serial.printf("[PLAYLIST] Unable to start PL%d\n", playlistNum);
    isMusicPlaying = false;
  } else {
    Serial.printf("[PLAYLIST] Started PL%d: %s\n", playlistNum, nowPlayingTrack.c_str());
  }
}

void playTrackByName(String filename) {
  filename.trim();
  if (filename.length() == 0) return;

  lastPlayedFile = "/music/" + filename;
  nowPlayingTrack = filename;
  lastPlayedTrackName = filename;
  currentPlaylistNum = 0;
  currentPlaylistTrackIndex = -1;
  currentPlaylistName = "DIRECT";
  isMusicPlaying = true;
  isSpotPlaying = false;
  manualDirectPlayback = true;
  userPaused = false;
  userStopped = false;
  audio.setVolume(currentMusicVolume);
  audio.connecttoFS(SD, lastPlayedFile.c_str());
  Serial.printf("[PLAY] Direct track: %s (vol=%d)\n", filename.c_str(), currentMusicVolume);
}

void playSpot(String filename, int volume, bool manualTrigger) {
  Serial.printf("[AUDIO] Playing spot: %s (vol=%d)%s\n", filename.c_str(), volume, manualTrigger ? " [manual]" : "");
  isSpotPlaying = true;
  manualSpotInProgress = manualTrigger;
  waitingSpotReplay = !manualTrigger;
  currentSpotVolume = volume;
  userPaused = false;
  audio.setVolume(currentSpotVolume);
  audio.connecttoFS(SD, ("/spot/" + filename).c_str());
}

void tryResumeScheduledOrSilence() {
  if (userStopped) {
    audio.stopSong();
    isMusicPlaying = false;
    currentPlaylistNum = 0;
    currentPlaylistTrackIndex = -1;
    nowPlayingTrack = "";
    currentPlaylistName = "";
    return;
  }

  time_t now;
  struct tm ti;
  time(&now);
  localtime_r(&now, &ti);
  int currentDay = (ti.tm_wday + 6) % 7;
  int currentMinute = ti.tm_hour * 60 + ti.tm_min;
  int scheduledPlaylist = findActivePlaylistForNow(currentDay, currentMinute);

  if (scheduledPlaylist > 0) {
    if (currentPlaylistNum != scheduledPlaylist) {
      playPlaylist(scheduledPlaylist, true);
    } else {
      if (!audio.isRunning()) {
        if (!playNextTrackInCurrentPlaylist(true)) {
          playPlaylist(scheduledPlaylist, true);
        }
      }
    }
  } else {
    audio.stopSong();
    isMusicPlaying = false;
    currentPlaylistNum = 0;
    currentPlaylistTrackIndex = -1;
    nowPlayingTrack = "";
    currentPlaylistName = "";
    Serial.println("[AUDIO] No scheduled playlist -> silence");
  }
}

void resumeMusic() {
  isSpotPlaying = false;
  manualSpotInProgress = false;
  waitingSpotReplay = false;

  if (userStopped) {
    audio.stopSong();
    isMusicPlaying = false;
    currentPlaylistNum = 0;
    currentPlaylistTrackIndex = -1;
    nowPlayingTrack = "";
    currentPlaylistName = "";
    Serial.println("[AUDIO] Resume blocked by user STOP -> silence");
    return;
  }

  tryResumeScheduledOrSilence();
}

void processScheduleEvents() {
  time_t nowTime;
  struct tm ti;
  time(&nowTime);
  localtime_r(&nowTime, &ti);
  int currentDay = (ti.tm_wday + 6) % 7;
  int currentMinute = ti.tm_hour * 60 + ti.tm_min;
  long currentEpochMinute = (long)(nowTime / 60);
  unsigned long nowMs = millis();

  int activePlaylistNum = -1;

  for (int i = 0; i < scheduleEventCount; i++) {
    ScheduleEvent& evt = scheduleEvents[i];
    if (evt.type != "playlist") continue;
    bool activeNow = isEventActiveNow(evt, currentDay, currentMinute);
    if (activeNow) activePlaylistNum = evt.target;

    if (activeNow && !evt.active) {
      evt.active = true;
      Serial.printf("[SCHEDULER] Playlist active: %s\n", getEventSummary(evt).c_str());
    } else if (!activeNow && evt.active) {
      evt.active = false;
      Serial.printf("[SCHEDULER] Playlist ended: %s\n", getEventSummary(evt).c_str());
    }
  }

  if (!isSpotPlaying && !userPaused && !userStopped && !transportTransition) {
    if (activePlaylistNum > 0) {
      if (currentPlaylistNum != activePlaylistNum) {
        playPlaylist(activePlaylistNum, true);
        lastActivePlaylistNum = activePlaylistNum;
      } else if (!audio.isRunning() && isMusicPlaying) {
        // if song ended between scheduler ticks, force next
        if (!playNextTrackInCurrentPlaylist(true)) {
          playPlaylist(activePlaylistNum, true);
        }
      }
    } else if (currentPlaylistNum > 0) {
      audio.stopSong();
      isMusicPlaying = false;
      currentPlaylistNum = 0;
      currentPlaylistTrackIndex = -1;
      nowPlayingTrack = "";
      Serial.println("[SCHEDULER] No active playlist - silence");
    }
  }

  int scheduledMusicVolume = defaultMusicVolume;
  bool volumeActive = false;
  for (int i = 0; i < scheduleEventCount; i++) {
    ScheduleEvent& evt = scheduleEvents[i];
    if (evt.type != "volume") continue;
    bool activeNow = isEventActiveNow(evt, currentDay, currentMinute);
    if (activeNow) {
      scheduledMusicVolume = evt.volume;
      volumeActive = true;
    }
    if (activeNow && !evt.active) evt.active = true;
    else if (!activeNow && evt.active) evt.active = false;
  }

  if (currentMusicVolume != scheduledMusicVolume) {
    currentMusicVolume = scheduledMusicVolume;
    if (!isSpotPlaying && isMusicPlaying) audio.setVolume(currentMusicVolume);
    Serial.printf("[SCHEDULER] Music volume set to %d%s\n", currentMusicVolume, volumeActive ? " (scheduled)" : " (default)");
  }

  int activeSpotIdx = -1;
  for (int i = 0; i < scheduleEventCount; i++) {
    if (scheduleEvents[i].type == "spot" && scheduleEvents[i].active) {
      activeSpotIdx = i;
      break;
    }
  }

  for (int i = 0; i < scheduleEventCount; i++) {
    ScheduleEvent& evt = scheduleEvents[i];
    if (evt.type != "spot") continue;
    if (evt.day != currentDay) continue;
    int triggerMinute = timeToMinutes(evt.startTime);
    if (triggerMinute < 0) continue;
    if (evt.active) {
      activeSpotIdx = i;
      continue;
    }
    if (activeSpotIdx >= 0 || isSpotPlaying) continue;
    if (currentMinute != triggerMinute) continue;
    if (evt.lastTriggerMinute == currentEpochMinute) continue;

    evt.lastTriggerMinute = currentEpochMinute;
    evt.pauseUntil = 0;
    evt.spotRepeatCount = 0;

    String spotFile = findSpotFile(evt.target);
    if (spotFile.length() == 0) {
      Serial.printf("[SCHEDULER] Spot file missing for Spot%d\n", evt.target);
      continue;
    }

    evt.active = true;
    playSpot(spotFile, evt.volume, false);
    evt.spotRepeatCount = 1;
    activeSpotIdx = i;
    Serial.printf("[SCHEDULER] Spot event started: %s\n", getEventSummary(evt).c_str());
  }

  // Handle scheduler-spot replay flow
  if (activeSpotIdx >= 0 && !audio.isRunning() && isSpotPlaying) {
    ScheduleEvent& evt = scheduleEvents[activeSpotIdx];

    if (evt.pauseUntil > 0) {
      if (nowMs >= evt.pauseUntil) {
        String spotFile = findSpotFile(evt.target);
        if (spotFile.length() > 0) {
          evt.pauseUntil = 0;
          playSpot(spotFile, evt.volume, false);
          evt.spotRepeatCount++;
          Serial.printf("[SCHEDULER] Spot replayed: %d/%d\n", evt.spotRepeatCount, evt.repeat);
        } else {
          evt.active = false;
          evt.pauseUntil = 0;
          evt.spotRepeatCount = 0;
          isSpotPlaying = false;
          waitingSpotReplay = false;
          resumeMusic();
        }
      }
    } else if (evt.spotRepeatCount < evt.repeat) {
      evt.pauseUntil = nowMs + 2000;
      waitingSpotReplay = true;
      Serial.printf("[SCHEDULER] Spot pause 2sec, replay %d/%d\n", evt.spotRepeatCount + 1, evt.repeat);
    } else {
      evt.active = false;
      evt.pauseUntil = 0;
      evt.spotRepeatCount = 0;
      isSpotPlaying = false;
      waitingSpotReplay = false;
      resumeMusic();
      Serial.println("[SCHEDULER] Spot finished all replays");
    }
  }

  // Handle manually triggered spot finishing
  if (manualSpotInProgress && isSpotPlaying && !audio.isRunning()) {
    isSpotPlaying = false;
    manualSpotInProgress = false;
    waitingSpotReplay = false;
    Serial.println("[AUDIO] Manual spot finished -> resume scheduled playback");
    resumeMusic();
  }
}

bool loadConfig() {
  File f = SD.open("/config/system.json");
  if (!f) return false;
  StaticJsonDocument<4096> doc;
  if (deserializeJson(doc, f)) {
    f.close();
    return false;
  }

  cfg.ssid = doc["wifi"]["ssid"] | "";
  cfg.pass = doc["wifi"]["pass"] | "";
  cfg.dhcp = doc["net"]["dhcp"] | true;
  cfg.ip = doc["net"]["ip"] | "192.168.1.100";
  cfg.gw = doc["net"]["gw"] | "192.168.1.1";
  cfg.mask = doc["net"]["mask"] | "255.255.255.0";
  cfg.dns = doc["net"]["dns"] | "8.8.8.8";
  cfg.mdns = doc["mdns"] | "jardio";
  cfg.musicVol = doc["volume"]["music"] | 15;
  cfg.spotVol = doc["volume"]["spot"] | 20;
  cfg.ntpServer = doc["time"]["ntp"] | "pool.ntp.org";
  cfg.tz = doc["time"]["tz"] | "CET-1CEST,M3.5.0,M10.5.0/3";

  defaultMusicVolume = cfg.musicVol;
  defaultSpotVolume = cfg.spotVol;
  currentMusicVolume = cfg.musicVol;
  currentSpotVolume = cfg.spotVol;

  JsonObject pmode = doc["playlistMode"];
  for (int i = 1; i <= 10; i++) {
    String kSh = "pl" + String(i) + "_shuffle";
    String kSt = "pl" + String(i) + "_start";
    playlistShuffle[i] = pmode[kSh] | false;
    playlistStartIndex[i] = pmode[kSt] | 0;
    if (playlistStartIndex[i] < 0) playlistStartIndex[i] = 0;
  }

  f.close();
  return true;
}

  SD.remove("/config/system.json");

  if (!f) return;

  StaticJsonDocument<4096> doc;
  doc["wifi"]["ssid"] = cfg.ssid;
  doc["wifi"]["pass"] = cfg.pass;

  doc["net"]["mask"] = cfg.mask;
  doc["net"]["dns"] = cfg.dns;
  doc["mdns"] = cfg.mdns;
  doc["volume"]["music"] = cfg.musicVol;
  doc["volume"]["spot"] = cfg.spotVol;
  doc["time"]["ntp"] = cfg.ntpServer;
  doc["time"]["tz"] = cfg.tz;

  JsonObject pmode = doc.createNestedObject("playlistMode");

  }

  f.close();
}

// ========== AUDIO TASKS ==========
void audioTask(void* pvParameters) {
  bool wasRunning = false;

  for (;;) {
    audio.loop();

    bool nowRunning = audio.isRunning();

    // reliable track-end progression
    if (wasRunning && !nowRunning) {
      if (!isSpotPlaying && !userPaused && !userStopped && !transportTransition && currentPlaylistNum > 0) {
        unsigned long nowMs = millis();
        if (nowMs - lastAutoNextMs > 200) {
          lastAutoNextMs = nowMs;
          if (!playNextTrackInCurrentPlaylist(true)) {
            // fallback to scheduler decision
            tryResumeScheduledOrSilence();
          }
        }
      } else if (!isSpotPlaying && manualDirectPlayback) {
        isMusicPlaying = false;
      }
    }

    wasRunning = nowRunning;
    vTaskDelay(pdMS_TO_TICKS(5));
  }
}

void schedulerTask(void* pvParameters) {
  for (;;) {
    processScheduleEvents();
    vTaskDelay(pdMS_TO_TICKS(100));
  }
}

void handleSavePlaylistsFor(WebServer& ws) {
  for (int i = 1; i <= 10; i++) {
    String filename = "/playlist/pl" + String(i) + ".csv";
    SD.remove(filename);
    File f = SD.open(filename, FILE_WRITE);
    if (f) {
      for (int j = 0; j < ws.args(); j++) {
        if (ws.argName(j) == "pl" + String(i)) {
          String val = urlDecode(ws.arg(j));
          f.println(val);
        }
      }
      f.close();
    }
  }
  ws.sendHeader("Location", "/playlists");
  ws.send(303);
}

void setPlaylistModeFromArgsFor(WebServer& ws) {
  for (int i = 1; i <= 10; i++) {
    String sh = "sh" + String(i);
    String st = "st" + String(i);
    playlistShuffle[i] = ws.hasArg(sh);
    if (ws.hasArg(st)) {
      int v = ws.arg(st).toInt();
      if (v < 0) v = 0;
      playlistStartIndex[i] = v;
    }
  }
  saveConfig();
}

void handleAddEventFor(WebServer& ws) {
  if (!ws.hasArg("day") || !ws.hasArg("startTime") || !ws.hasArg("type")) {
    ws.send(400, "text/plain", "Missing required args");
    return;
  }
  if (scheduleEventCount >= 100) {
    ws.send(400, "text/plain", "Scheduler full");
    return;
  }

  ScheduleEvent evt;
  evt.id = generateEventId();
  evt.day = ws.arg("day").toInt();
  evt.startTime = ws.arg("startTime");
  evt.endTime = "";
  evt.type = ws.arg("type");
  evt.target = 0;
  evt.repeat = 1;
  evt.volume = defaultMusicVolume;
  resetEventRuntimeState(evt);

  if (evt.type == "playlist") {
    if (!ws.hasArg("endTime") || !ws.hasArg("playlistTarget")) {
      ws.send(400, "text/plain", "Playlist requires endTime and playlistTarget");
      return;
    }
    evt.endTime = ws.arg("endTime");
    evt.target = ws.arg("playlistTarget").toInt();
  } else if (evt.type == "volume") {
    if (!ws.hasArg("endTime") || !ws.hasArg("volumeValue")) {
      ws.send(400, "text/plain", "Volume requires endTime and volumeValue");
      return;
    }
    evt.endTime = ws.arg("endTime");
    evt.volume = ws.arg("volumeValue").toInt();
  } else if (evt.type == "spot") {
    if (!ws.hasArg("spotTarget")) {
      ws.send(400, "text/plain", "Spot requires spotTarget");
      return;
    }
    evt.target = ws.arg("spotTarget").toInt();
    evt.repeat = ws.hasArg("repeat") ? ws.arg("repeat").toInt() : 1;
    evt.volume = ws.hasArg("spotVolume") ? ws.arg("spotVolume").toInt() : defaultSpotVolume;
  } else {
    ws.send(400, "text/plain", "Invalid event type");
    return;
  }

  evt.repeat = constrain(evt.repeat, 1, 5);
  evt.volume = constrain(evt.volume, 0, 21);
  if (evt.type == "playlist" || evt.type == "spot") evt.target = constrain(evt.target, 1, 10);
  if (evt.day < 0 || evt.day > 6 || !isValidTimeRangeForEvent(evt)) {
    ws.send(400, "text/plain", "Invalid event data");
    return;
  }

  scheduleEvents[scheduleEventCount++] = evt;
  saveScheduleEvents();
  ws.sendHeader("Location", "/scheduler");
  ws.send(303);
}

void handleEditEventFor(WebServer& ws) {
  if (!ws.hasArg("id")) {
    ws.send(404);
    return;
  }
  String eventId = ws.arg("id");

  for (int i = 0; i < scheduleEventCount; i++) {
    if (scheduleEvents[i].id == eventId) {
      ScheduleEvent updated = scheduleEvents[i];
      if (ws.hasArg("day")) updated.day = ws.arg("day").toInt();
      if (ws.hasArg("startTime")) updated.startTime = ws.arg("startTime");
      if (updated.type == "playlist" || updated.type == "volume") {
        if (ws.hasArg("endTime")) updated.endTime = ws.arg("endTime");
      } else updated.endTime = "";
      if (ws.hasArg("target")) updated.target = ws.arg("target").toInt();
      if (ws.hasArg("repeat")) updated.repeat = constrain(ws.arg("repeat").toInt(), 1, 5);
      if (ws.hasArg("volume")) updated.volume = constrain(ws.arg("volume").toInt(), 0, 21);
      if (updated.type == "playlist" || updated.type == "spot") updated.target = constrain(updated.target, 1, 10);

      if (updated.day < 0 || updated.day > 6 || !isValidTimeRangeForEvent(updated)) {
        ws.send(400, "text/plain", "Invalid event data");
        return;
      }

      resetEventRuntimeState(updated);
      scheduleEvents[i] = updated;
      saveScheduleEvents();
      ws.sendHeader("Location", "/scheduler");
      ws.send(303);
      return;
    }
  }
  ws.send(404);
}

void handleDeleteEventFor(WebServer& ws) {
  if (!ws.hasArg("id")) {
    ws.send(404);
    return;
  }
  String eventId = ws.arg("id");

  for (int i = 0; i < scheduleEventCount; i++) {
    if (scheduleEvents[i].id == eventId) {
      for (int j = i; j < scheduleEventCount - 1; j++) scheduleEvents[j] = scheduleEvents[j + 1];
      scheduleEventCount--;
      saveScheduleEvents();
      ws.send(200);
      return;
    }
  }
  ws.send(404);
}

void handleCopyEventFor(WebServer& ws) {
  if (!(ws.hasArg("id") && ws.hasArg("days"))) {
    ws.send(400);
    return;
  }

  String eventId = ws.arg("id");
  String daysStr = ws.arg("days");
  int srcIdx = -1;
  for (int i = 0; i < scheduleEventCount; i++) {
    if (scheduleEvents[i].id == eventId) {
      srcIdx = i;
      break;
    }
  }
  if (srcIdx < 0) {
    ws.send(404);
    return;
  }

  int days[7], dayCount = 0;
  String currentNum = "";
  for (int i = 0; i <= daysStr.length(); i++) {
    char c = (i < daysStr.length()) ? daysStr[i] : ',';
    if (c == ',') {
      if (currentNum.length() > 0 && dayCount < 7) {
        days[dayCount++] = currentNum.toInt();
        currentNum = "";
      }
    } else currentNum += c;
  }

  bool selectedDays[7] = { false, false, false, false, false, false, false };
  for (int d = 0; d < dayCount; d++)
    if (days[d] >= 0 && days[d] < 7) selectedDays[days[d]] = true;

  ScheduleEvent src = scheduleEvents[srcIdx];
  for (int day = 0; day < 7; day++) {
    if (!selectedDays[day]) continue;
    if (day == src.day) continue;

    bool duplicate = false;
    for (int i = 0; i < scheduleEventCount; i++) {
      if (eventsEquivalentForDay(src, scheduleEvents[i], day)) {
        duplicate = true;
        break;
      }
    }
    if (duplicate) continue;
    if (scheduleEventCount >= 100) break;

    ScheduleEvent& newEvt = scheduleEvents[scheduleEventCount];
    newEvt = src;
    newEvt.id = generateEventId();
    newEvt.day = day;
    resetEventRuntimeState(newEvt);
    scheduleEventCount++;
  }

  saveScheduleEvents();
  ws.sendHeader("Location", "/scheduler");
  ws.send(303);
}

void setup() {
  Serial.begin(115200);
  delay(500);
  Serial.printf("\n\n[BOOT] Jardio OS %s starting...\n", FIRMWARE_VERSION);

  sdSPI.begin(SD_SCK, SD_MISO, SD_MOSI, SD_CS);
  if (!SD.begin(SD_CS, sdSPI, 20000000)) {
    Serial.println("[ERROR] SD card init FAILED!");
    while (1) delay(1000);
  }
  Serial.println("[OK] SD card initialized");

  initPlaylists();
  initScheduler();

  const char* dirs[] = { "/config", "/music", "/playlist", "/scheduler", "/spot" };
  for (const char* d : dirs)
    if (!SD.exists(d)) SD.mkdir(d);

  indexTracks();
  loadConfig();
  loadScheduleEvents();

  Serial.printf("[CONFIG] Loaded - Music Vol: %d, Spot Vol: %d\n", defaultMusicVolume, defaultSpotVolume);

  if (cfg.ssid.length() > 0) {
    WiFi.mode(WIFI_STA);
    if (!cfg.dhcp) {
      IPAddress ip, gw, nm, d1;
      ip.fromString(cfg.ip);
      gw.fromString(cfg.gw);
      nm.fromString(cfg.mask);
      d1.fromString(cfg.dns);
      WiFi.config(ip, gw, nm, d1);
    }
    WiFi.begin(cfg.ssid.c_str(), cfg.pass.c_str());
    unsigned long st = millis();
    while (WiFi.status() != WL_CONNECTED && millis() - st < 8000) delay(500);
  }

  if (WiFi.status() != WL_CONNECTED) {
    apMode = true;
    WiFi.mode(WIFI_AP);
    WiFi.softAP("Jardio-SP1", "admin123");
    Serial.println("[WIFI] AP Mode: Jardio-SP1 / admin123");
  } else {
    Serial.printf("[WIFI] Connected: %s - IP: %s\n", cfg.ssid.c_str(), WiFi.localIP().toString().c_str());
    MDNS.begin(cfg.mdns.c_str());
    configTime(0, 0, cfg.ntpServer.c_str());
    setenv("TZ", cfg.tz.c_str(), 1);
    tzset();
  }

  audio.setPinout(I2S_BCLK, I2S_LRC, I2S_DOUT);
  audio.setVolume(currentMusicVolume);
  Serial.println("[AUDIO] I2S configured");

  xTaskCreatePinnedToCore(audioTask, "AudioTask", 20000, NULL, 6, NULL, 1);
  xTaskCreatePinnedToCore(schedulerTask, "SchedulerTask", 5000, NULL, 4, &schedulerTaskHandle, 0);
  Serial.println("[TASKS] Created");

  // ===== UI ROUTES (PORT 80) =====
  server.on("/", []() {
    server.send(200, "text/html", apMode ? getSetupUI() : getDashboard());
  });
  server.on("/media", []() {
    server.send(200, "text/html", getMediaUI());
  });
  server.on("/playlists", []() {
    server.send(200, "text/html", getPlaylistUI());
  });
  server.on("/scheduler", []() {
    server.send(200, "text/html", getSchedulerUI());
  });
  server.on("/system", []() {
    server.send(200, "text/html", getSystemUI());
  });

  // ===== API ROUTES (PORT 11000) =====
  apiServer.on("/api/saveplaylists", HTTP_POST, []() {
    handleSavePlaylistsFor(apiServer);
  });

  apiServer.on("/api/playlistmode", HTTP_POST, []() {
    setPlaylistModeFromArgsFor(apiServer);
    apiServer.sendHeader("Location", "/playlists");
    apiServer.send(303);
  });

  apiServer.on("/api/event/add", HTTP_POST, []() {
    handleAddEventFor(apiServer);
  });
  apiServer.on("/api/event/edit", HTTP_POST, []() {
    handleEditEventFor(apiServer);
  });
  apiServer.on("/api/event/delete", []() {
    handleDeleteEventFor(apiServer);
  });
  apiServer.on("/api/event/copy", []() {
    handleCopyEventFor(apiServer);
  });

  apiServer.on("/api/play", []() {
    if (apiServer.hasArg("f")) {
      String fName = urlDecode(apiServer.arg("f"));
      userStopped = false;
      pausedByUserWithTrack = false;
      pausedTrackFile = "";
      pausedPlaylistNum = 0;
      pausedPlaylistIndex = -1;
      playTrackByName(fName);
    }
    apiServer.send(200);
  });

  apiServer.on("/api/play-spot", []() {
    if (!apiServer.hasArg("f")) {
      apiServer.send(400, "text/plain", "Missing arg f");
      return;
    }

    String raw = urlDecode(apiServer.arg("f"));
    raw.trim();

    int vol = apiServer.hasArg("vol") ? apiServer.arg("vol").toInt() : defaultSpotVolume;
    vol = constrain(vol, 0, 21);

    String fName = "";
    bool numeric = raw.length() > 0;
    for (unsigned int i = 0; i < raw.length(); i++) {
      if (!isDigit(raw.charAt(i))) {
        numeric = false;
        break;
      }
    }

    if (numeric) {
      int spotNum = raw.toInt();
      if (spotNum >= 1 && spotNum <= 10) fName = findSpotFile(spotNum);
    }

    if (fName.length() == 0) {
      apiServer.send(404, "text/plain", "Spot file not found");
      return;
    }

    nowPlayingTrack = fName;
    playSpot(fName, vol, true);
    apiServer.send(200, "text/plain", "OK");
  });

  apiServer.on("/api/pause", []() {
    if (!userPaused && audio.isRunning()) {
      transportTransition = true;

      pausedTrackFile = lastPlayedFile;
      pausedPlaylistNum = currentPlaylistNum;
      pausedPlaylistIndex = currentPlaylistTrackIndex;
      pausedByUserWithTrack = (pausedTrackFile.length() > 0);

      audio.stopSong();
      userPaused = true;
      userStopped = false;
      isMusicPlaying = false;

      transportTransition = false;
      Serial.println("[API] PAUSE");
    }
    apiServer.send(200);
  });

  apiServer.on("/api/resume", []() {
    if (userPaused) {
      transportTransition = true;
      userPaused = false;
      userStopped = false;

      if (pausedByUserWithTrack && pausedTrackFile.length() > 0) {
        audio.setVolume(currentMusicVolume);
        audio.connecttoFS(SD, pausedTrackFile.c_str());

        isMusicPlaying = true;
        manualDirectPlayback = (pausedPlaylistNum == 0);
        currentPlaylistNum = pausedPlaylistNum;
        currentPlaylistTrackIndex = pausedPlaylistIndex;

        int slash = pausedTrackFile.lastIndexOf('/');
        nowPlayingTrack = (slash >= 0) ? pausedTrackFile.substring(slash + 1) : pausedTrackFile;

        pausedByUserWithTrack = false;
        pausedTrackFile = "";
        pausedPlaylistNum = 0;
        pausedPlaylistIndex = -1;

        Serial.println("[API] RESUME (same track)");
      } else {
        tryResumeScheduledOrSilence();
        Serial.println("[API] RESUME (scheduler)");
      }

      transportTransition = false;
    } else if (userStopped) {
      transportTransition = true;
      userStopped = false;
      userPaused = false;
      tryResumeScheduledOrSilence();
      transportTransition = false;
      Serial.println("[API] RESUME (after STOP)");
    } else if (!audio.isRunning()) {
      transportTransition = true;
      tryResumeScheduledOrSilence();
      transportTransition = false;
      Serial.println("[API] RESUME (scheduler)");
    }
    apiServer.send(200);
  });

  apiServer.on("/api/stop", []() {
    audio.stopSong();
    userPaused = false;
    userStopped = true;
    isSpotPlaying = false;
    manualSpotInProgress = false;
    waitingSpotReplay = false;
    isMusicPlaying = false;
    manualDirectPlayback = false;
    currentPlaylistNum = 0;
    currentPlaylistTrackIndex = -1;
    nowPlayingTrack = "";
    currentPlaylistName = "";

    pausedByUserWithTrack = false;
    pausedTrackFile = "";
    pausedPlaylistNum = 0;
    pausedPlaylistIndex = -1;
    transportTransition = false;

    Serial.println("[API] STOP");
    apiServer.send(200);
  });

  apiServer.on("/api/vol", HTTP_POST, []() {
    if (apiServer.hasArg("musicvol")) {
      defaultMusicVolume = constrain(apiServer.arg("musicvol").toInt(), 0, 21);
      currentMusicVolume = defaultMusicVolume;
      cfg.musicVol = defaultMusicVolume;
    }
    if (apiServer.hasArg("spotvol")) {
      defaultSpotVolume = constrain(apiServer.arg("spotvol").toInt(), 0, 21);
      currentSpotVolume = defaultSpotVolume;
      cfg.spotVol = defaultSpotVolume;
    }
    saveConfig();
    apiServer.sendHeader("Location", "/system");
    apiServer.send(303);
  });

  apiServer.on("/api/time", HTTP_POST, []() {
    if (apiServer.hasArg("ntp")) cfg.ntpServer = apiServer.arg("ntp");
    if (apiServer.hasArg("tz")) cfg.tz = apiServer.arg("tz");
    saveConfig();
    configTime(0, 0, cfg.ntpServer.c_str());
    setenv("TZ", cfg.tz.c_str(), 1);
    tzset();
    apiServer.sendHeader("Location", "/system");
    apiServer.send(303);
  });

  apiServer.on("/api/net", HTTP_POST, []() {
    cfg.dhcp = (apiServer.arg("dhcp") == "1");
    cfg.ip = apiServer.arg("ip");
    cfg.gw = apiServer.arg("gw");
    cfg.mask = apiServer.arg("mask");
    cfg.dns = apiServer.arg("dns");
    saveConfig();
    apiServer.sendHeader("Location", "/system");
    apiServer.send(303);
  });

  apiServer.on("/api/restart", []() {
    apiServer.send(200);
    delay(500);
    ESP.restart();
  });

  apiServer.on("/api/delete", []() {
    if (apiServer.hasArg("p")) {
      SD.remove(urlDecode(apiServer.arg("p")));
      indexTracks();
    }
    apiServer.send(200);
  });

  server.begin();
  apiServer.begin();
  Serial.println("[OK] Web server started on :80");
  Serial.println("[OK] API server started on :11000");
  Serial.printf("[BOOT] Jardio OS %s READY!\n", FIRMWARE_VERSION);
}
